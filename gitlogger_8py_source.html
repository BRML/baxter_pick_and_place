27 # https://github.com/surban/mlutils/commit/1770edb54bdb04b90970705c1
28 
29 import importlib
30 import json
31 import os
32 from subprocess import Popen, PIPE
33 
34 
35 def git_logger(modules=None, filename=None):
36  """Versioning logger that logs the current state of the requested modules.
37  If module is provided as a git repository, this state is represented by
38  the latest git commit.
39  If module is not provided as a git repository, the version info of the
40  module is stored, if available.
41 
42  Usage: print git_logger(['numpy', 'theano'])
43  {
44  "commits": {
45  "theano": "0693ce052725a15b502068a1490b0637216feb00",
46  "numpy": "1.8.2"
47  },
48  "warnings": {
49  "theano": ""
50  }
51  }
52 
53  :param modules: A list of modules to check for version information.
54  :param filename: If given, the filename to save the log with.
55  :return: The created log as a dictionary.
56  """
57  if modules is None:
58  modules = list()
59  if not isinstance(modules, list):
60  modules = [modules, ]
61  current_path = os.getcwd()
62  commits = dict()
63  warnings = dict()
64 
65  for module in modules:
66  # get install path of module and go there
67  if module.startswith('ros_'):
68  import rospkg
69  module = module.split('_', 1)[1]
70  mod_path = rospkg.RosPack().get_path(module)
71  else:
72  mod = importlib.import_module(module)
73  mod_path = os.path.dirname(mod.__file__)
74  os.chdir(mod_path)
75 
76  proc = Popen(['git', 'log', '-1', '--pretty=oneline'],
77  stdout=PIPE, stderr=PIPE)
78  out, err = proc.communicate()
79  if err.startswith('fatal'):
80  # mod_path is not a git repository.
81  # Fallback to module version instead.
82  if hasattr(mod, '__version__'):
83  commits[module] = mod.__version__
84  else:
85  # mod_path is neither a git repository
86  # nor has module a version info => give up
87  commits[module] = 'no commit or version information found!'
88  else:
89  commits[module] = out.split(' ')[0].strip()
90  # give a warning if there have been changes since latest commit
91  proc = Popen(['git', 'diff', '--shortstat'], stdout=PIPE)
92  out, _ = proc.communicate()
93  if out:
94  warnings[module] = out
95  warn = "WARNING: uncommitted changes in module '{}'!".format(module)
96  print "\033[1;33m%s\033[0m\n" % warn
97  logs = {'commits': commits, 'warnings': warnings}
98  os.chdir(current_path)
99 
100  # save versioning log, if requested
101  if isinstance(filename, str) and len(filename) > 0:
102  _, ext = os.path.splitext(filename)
103  if not ext:
104  filename += '.json'
105  with open(filename, 'wb') as fp:
106  json.dump(logs, fp, indent=4)
107  return logs
