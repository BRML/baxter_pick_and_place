101  def _im_detect(self, image):
102  """Taken from
103  https://github.com/daijifeng001/MNC/blob/master/tools/demo.py.
104  Somehow combines different stages of the network. No idea how it works.
105 
106  :param image: An image (numpy array) of shape (height, width, 3).
107  :return: A tuple of three numpy arrays, the n_proposals x n_classes
108  scores, the corresponding n_proposals x 4 bounding boxes, where
109  each bounding box is defined as <xul, yul, xlr, ylr> and the
110  n_proposals x 1 x 21 x 21 segmentation masks.
111  """
112  forward_kwargs, im_scales = self._prepare_mnc_args(image)
113  blobs_out = self._net.forward(**forward_kwargs)
114  # output we need to collect:
115  # 1. output from phase1'
116  rois_phase1 = self._net.blobs['rois'].data.copy()
117  masks_phase1 = self._net.blobs['mask_proposal'].data[...]
118  scores_phase1 = self._net.blobs['seg_cls_prob'].data[...]
119  # 2. output from phase2
120  rois_phase2 = self._net.blobs['rois_ext'].data[...]
121  masks_phase2 = self._net.blobs['mask_proposal_ext'].data[...]
122  scores_phase2 = self._net.blobs['seg_cls_prob_ext'].data[...]
123  # Boxes are in resized space, we un-scale them back
124  rois_phase1 = rois_phase1[:, 1:5] / im_scales[0]
125  rois_phase2 = rois_phase2[:, 1:5] / im_scales[0]
126  rois_phase1, _ = clip_boxes(rois_phase1, image.shape)
127  rois_phase2, _ = clip_boxes(rois_phase2, image.shape)
128  # concatenate two stages to get final network output
129  masks = np.concatenate((masks_phase1, masks_phase2), axis=0)
130  boxes = np.concatenate((rois_phase1, rois_phase2), axis=0)
131  scores = np.concatenate((scores_phase1, scores_phase2), axis=0)
132  return scores, boxes, masks

153  def detect(self, image):
154  """Feed forward the given image through the previously loaded network.
155  Return scores, bounding boxes and segmentation masks for all abject
156  proposals and classes.
157 
158  :param image: An image (numpy array) of shape (height, width, 3).
159  :return: A tuple of three numpy arrays, the n_proposals x n_classes
160  scores, the corresponding n_proposals x 4 bounding boxes, where
161  each bounding box is defined as <xul, yul, xlr, ylr> and the
162  n_proposals x 1 x 21 x 21 segmentation masks.
163  """
164  if self._net is None:
165  raise RuntimeError("No loaded network found! "
166  "Did you run init_model()?")
167  if len(image.shape) != 3 and image.shape[2] != 3:
168  raise ValueError("Image must be a three channel color image "
169  "with shape (h, w, 3)!")
170  start = time.time()
171  scores, boxes, masks = self._im_detect(image)
172  self._logger.debug('Detection took {:.3f}s for {:d} object proposals'.format(
173  time.time() - start, boxes.shape[0])

178  def _mask_image_from_mask(mask, box, image_size):
179  """Convert an MNC mask detection into a binary mask image.
180 
181  :param mask: The MNC mask detection to convert (a 21x21 numpy array).
182  :param box: The corresponding bounding box of the detection.
183  :param image_size: The height and width of the input and mask images.
184  :return: The segmentation of the detection; a (height, width) numpy
185  array.
186  """
187  def clip_box(box, image_size):
188  """Clip bounding box to image size.
189 
190  :param box: The bounding box, defined as <xul, yul, xlr, ylr>.
191  :param image_size: The height and width of the image.
192  :return: The bounding box coordinates, clipped to the image size
193  and rounded to the next integer value.
194  """
195  # clip box into image space
196  box = np.round(box).astype(np.uint32)
197  height, width = image_size
198  box[0] = min(max(box[0], 0), width - 1)
199  box[1] = min(max(box[1], 0), height - 1)
200  box[2] = min(max(box[2], 0), width - 1)
201  box[3] = min(max(box[3], 0), height - 1)
202  return box
203 
204  height, width = image_size
205  mask_img = np.zeros((height, width), dtype=np.uint8)
206  box = clip_box(box=box, image_size=image_size)
207  mask = cv2.resize(mask, (box[2] - box[0] + 1, box[3] - box[1] + 1))
208  mask = mask >= cfg.BINARIZE_THRESH
209  mask = 255*np.array(mask, dtype=np.uint8)
210  mask_img[box[1]:box[3] + 1, box[0]:box[2] + 1] = mask
211  return mask_img

213  def detect_object(self, image, object_id, threshold=0.5):
214  """Feed forward the given image through the previously loaded network.
215  Return the bounding box and segmentation with the highest score for
216  the requested object class.
217 
218  :param image: An image (numpy array) of shape (height, width, 3).
219  :param object_id: One object identifier string contained in the list
220  of objects.
221  :param threshold: The threshold (0, 1) on the score for a detection
222  to be considered as valid.
223  :return: A dictionary containing the detection with
224  'id': The object identifier.
225  'score: The score of the detection (scalar).
226  'box': The bounding box of the detection; a (4,) numpy array.
227  'mask': The segmentation of the detection; a (height, width)
228  numpy array.
229  """
230  if object_id not in self._classes:
231  raise KeyError("Object {} is not contained in the defined "
232  "set of objects!".format(object_id))
233  scores, boxes, masks = self.detect(image=image)
234 
235  # Find scores for requested object class
236  cls_idx = self._classes.index(object_id)
237  cls_scores = scores[:, cls_idx]
238 
239  best_idx = np.argmax(cls_scores)
240  best_score = cls_scores[best_idx]
241  best_box = boxes[best_idx]
242  best_mask = self._mask_image_from_mask(mask=masks[best_idx][0],
243  box=best_box,
244  image_size=image.shape[:2])
245  self._logger.debug('Best score for {} is {:.3f} {} {:.3f}'.format(
246  object_id,
247  best_score,
248  '>=' if best_score > threshold else '<',
249  threshold)
250  )
251  if best_score > threshold:
252  return {'id': object_id, 'score': best_score, 'box': best_box, 'mask': best_mask}
253  return {'id': object_id, 'score': best_score, 'box': None, 'mask': None}

255  def detect_best(self, image, threshold=0.5):
256  """Feed forward the given image through the previously loaded network.
257  Return the bounding box and segmentation with the highest score
258  amongst all classes.
259 
260  :param image: An image (numpy array) of shape (height, width, 3).
261  :param threshold: The threshold (0, 1) on the score for a detection
262  to be considered as valid.
263  :return: A dictionary containing the detection with
264  'id': The object identifier.
265  'score: The score of the detection (scalar).
266  'box': The bounding box of the detection; a (4,) numpy array.
267  'mask': The segmentation of the detection; a (height, width)
268  numpy array.
269  """
270  scores, boxes, masks = self.detect(image=image)
271 
272  # find best score among all classes (except background)
273  best_proposal, best_class = np.unravel_index(scores[:, 1:].argmax(),
274  scores[:, 1:].shape)
275  best_class += 1 # compensate for background
276  best_score = scores[best_proposal, best_class]
277  best_box = boxes[best_proposal]
278  best_mask = self._mask_image_from_mask(mask=masks[best_proposal][0],
279  box=best_box,
280  image_size=image.shape[:2])
281  best_object = self._classes[best_class]
282 
283  self._logger.debug('Best score for {} is {:.3f} {} {:.3f}'.format(
284  best_object,
285  best_score,
286  '>=' if best_score > threshold else '<',
287  threshold)
288  )
289  if best_score > threshold:
290  return {'id': best_object, 'score': best_score, 'box': best_box, 'mask': best_mask}
291  return {'id': best_object, 'score': best_score, 'box': None, 'mask': None}
