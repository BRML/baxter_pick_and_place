38 class Camera(object):
39  def __init__(self, topic, prefix, cam_pars=None):
40  """Base class for a ROS camera.
41  A camera should have at least
42  - a method to read images from a ROS topic,
43  - a method to project pixel coordinates to camera coordinates, and
44  - a method to project camera coordinates to pixel coordinates.
45 
46  :param topic: The ROS image topic to read camera images from.
47  :param prefix: The prefix for the logger name to use.
48  :param cam_pars: An optional dictionary containing camera parameters.
49  If given it replaces the parameters read from the camera driver
50  via ROS and is required to contain
51  - a 3x3 camera matrix,
52  - the image size (height, width) and
53  - the camera distortion coefficients.
54  """
55  self._topic = topic
56 
57  self._logger = logging.getLogger('{}.cam'.format(prefix))
58 
59  self.camera_matrix = None
60  self.image_size = None
61  self.distortion_coeff = None
62  if cam_pars is None:
63  self._get_ros_calibration()
64  else:
65  self.camera_matrix = cam_pars['cam_mat']
66  self.image_size = tuple(cam_pars['size'])
67  self.distortion_coeff = cam_pars['dist_coeff']
68  if self.camera_matrix.shape != (3, 3):
69  raise ValueError("Expected a 3x3 camera matrix, got "
70  "{}!".format(self.camera_matrix.shape))
71 
72  self.meters_per_pixel = None

74  def _get_ros_calibration(self):
75  """Read the calibration data of the camera from the ROS topic. For
76  additional information see
77  http://docs.ros.org/indigo/api/sensor_msgs/html/msg/CameraInfo.html.
78 
79  :return:
80  """
81  topic = self._topic.rsplit('/', 1)[0] + '/camera_info'
82  self._logger.info("Try to read camera info from {}.".format(topic))
83  try:
84  # try to read calibration from ROS camera info topic
85  msg = rospy.wait_for_message(topic=topic, topic_type=CameraInfo,
86  timeout=1.5)
87  self.camera_matrix = np.asarray(msg.K, dtype=np.float64).reshape((3, 3))
88  self.image_size = (int(msg.height), int(msg.width))
89  self.distortion_coeff = np.asarray(msg.D, dtype=np.float64)
90  except rospy.ROSException:
91  raise RuntimeError("Unable to read camera info from ROS master!")

93  def collect_image(self):
94  """Read the most recent image message from the ROS topic and convert
95  it into a numpy array.
96 
97  :return: An image (a (height, width, n_channels) numpy array).
98  """
99  try:
100  msg = rospy.wait_for_message(topic=self._topic,
101  topic_type=Image,
102  timeout=0.5)
103  img = imgmsg_to_img(imgmsg=msg)
104  except rospy.ROSException:
105  msg = "ROS error while reading image from {}.".format(self._topic)
106  self._logger.error(msg)
107  raise RuntimeError(msg)
108  if img.dtype == np.float32:
109  # In simulation, depth map is a float32 image
110  mask = np.isnan(img)
111  if mask.any():
112  # In simulation, the background has NaN depth values.
113  # We replace them with 0 m, similar to what the Kinect V1 did.
114  # See https://msdn.microsoft.com/en-us/library/jj131028.aspx.
115  self._logger.debug("{}: There was at least one NaN in the depth "
116  "image. I replaced all occurrences with "
117  "0.0 m.".format(self._topic))
118  img.flags.writeable = True
119  img[mask] = 0.0
120  # We now map the float values in meters to uint16 values in mm
121  # as provided by the libfreenect2 library and Kinect SDK.
122  img *= 1000.0
123  img = img.astype(np.uint16, copy=False)
124  return img

126  def projection_pixel_to_camera(self, pixel, z):
127  """Project a 2d point (px, py) in pixel coordinates into 3D camera
128  coordinates.
129  Since the projection from camera to pixel space is non-invertible,
130  we are restricted to the case that we know the z coordinate of the
131  point in camera coordinates, e.g., for an object lying on a table
132  with known height.
133 
134  :param pixel: A 2D position as a tuple of length 2 (px, py).
135  :param z: The known height of the pixel in camera coordinates.
136  :return: The corresponding 3D camera coordinates [x, y, z].
137  """
138  if isinstance(pixel, (tuple, list)) and len(pixel) == 2:
139  # For known z coordinate we can rearrange
140  # (u) [fx 0 cx](x)
141  # (v) = [0 fy cy](y)
142  # (w) [0 0 1 ](z)
143  # to compute
144  # w = z,
145  # x = (u - cx*z)/fx, u = px*w, and
146  # y = (v - cy*z)/fy, v = py*w.
147  x = z * (pixel[0] - self.camera_matrix[0, 2]) / self.camera_matrix[0, 0]
148  y = z * (pixel[1] - self.camera_matrix[1, 2]) / self.camera_matrix[1, 1]
149  # flip y axis
150  return [x, -y, z]
151  raise ValueError("'pixel' should be a tuple of length 2!")

153  def projection_camera_to_pixel(self, position):
154  """Project a 3d point [x, y, z] in camera coordinates onto the
155  rectified image. For additional information see
156  http://docs.ros.org/indigo/api/sensor_msgs/html/msg/CameraInfo.html.
157 
158  :param position: A 3D position as a list of length 3 [x, y, z].
159  :return: The corresponding pixel coordinates (px, py).
160  """
161  if isinstance(position, list) and len(position) == 3:
162  u, v, w = np.dot(self.camera_matrix, np.asarray(position))
163  px = float(u)/w
164  py = float(v)/w
165  # flip image in y direction
166  return px, self.image_size[0] - py
167  raise ValueError("'position' should be a list of length 3!")

170 def imgmsg_to_img(imgmsg):
171  """Convert a ROS image message to a numpy array holding the image.
172 
173  :param imgmsg: A ROS image message.
174  :return: The BGR image as a (height, width, n_channels) numpy array.
175  """
176  img = None
177  for enc in ['bgr8', 'mono8', 'passthrough']:
178  try:
179  img = cv_bridge.CvBridge().imgmsg_to_cv2(imgmsg, enc)
180  except cv_bridge.CvBridgeError:
181  pass
182  except AttributeError as e:
183  raise e
184  if img is not None:
185  break
186  if img is None:
187  raise ValueError("Cannot convert image message to numpy array!")
188  return img
