194  def estimate_hand_trafo(self, n):
195  """Estimate the transformation between Baxter's end effector and the
196  origin of the calibration pattern mounted on it using the algorithm
197  by Tsai and Lenz.
198 
199  :param n: The number of absolute end effector poses to use.
200  :return: The homogeneous transform between TCP and the origin of the
201  calibration pattern.
202  """
203  if n < 3:
204  raise ValueError("At least 3 poses are needed to compute the "
205  "transformation!")
206  btt = list()
207  cto = list()
208  pattern = self._pattern.T[:, np.newaxis, :]
209 
210  self.logger.info("Record %d absolute pose and pattern pairs." % n)
211  while len(btt) < n and not rospy.is_shutdown():
212  if self._move():
213  rospy.sleep(1.0)
214  bttn = self._robot.hom_gripper_to_robot(arm=self._arm)
215  color, _, _ = self._kinect.collect_data(color=True)
216 
217  self._pub_vis.publish(img_to_imgmsg(img=color))
218  patternfound, centers = cv2.findCirclesGridDefault(
219  image=color, patternSize=self._patternsize,
220  flags=cv2.CALIB_CB_ASYMMETRIC_GRID)
221  if patternfound == 0:
222  self.logger.debug("No pattern found!")
223  continue
224 
225  rot, trans, _ = cv2.solvePnPRansac(
226  objectPoints=pattern, imagePoints=centers,
227  cameraMatrix=self._kinect.color.camera_matrix,
228  distCoeffs=self._kinect.color.distortion_coeff,
229  flags=cv2.CV_ITERATIVE)
230  rot, _ = cv2.Rodrigues(rot)
231  cton = np.eye(4)
232  cton[:-1, :-1] = rot
233  cton[:-1, -1] = np.squeeze(trans)
234 
235  fname = os.path.join(self._sink, "1_tto_%d" % (len(btt) + 1))
236  cv2.imwrite(fname + ".jpg", color)
237  cv2.drawChessboardCorners(image=color,
238  patternSize=self._patternsize,
239  corners=centers,
240  patternWasFound=patternfound)
241  self._pub_vis.publish(img_to_imgmsg(img=color))
242  cv2.imwrite(fname + "_det.jpg", color)
243 
244  btt.append(bttn)
245  fname = os.path.join(self._sink, "1_btt_%d.npz" % len(btt))
246  np.savez(fname, bttn=btt[-1])
247  cto.append(cton)
248  fname = os.path.join(self._sink, "1_cto_%d.npz" % len(cto))
249  np.savez(fname, cton=cto[-1])
250 
251  self.logger.info("Compute %d relative transforms." % ((n**2 - n)/2))
252  bttij = list()
253  ctoij = list()
254  for i in range(n):
255  for j in range(i + 1, n):
256  bttij.append(np.dot(inv_trafo_matrix(btt[j]), btt[i]))
257  ctoij.append(np.dot(cto[j], inv_trafo_matrix(cto[i])))
258 
259  self.logger.info('Apply algorithm by Tsai and Lenz (1989).')
260  tto = tsai_lenz_89(a=bttij, b=ctoij)
261  fname = os.path.join(self._sink, "1_tto.npz")
262  np.savez(fname, tto=tto)
263  return tto

265  def estimate_cam_trafo(self, n, tto):
266  """Estimate the transformation between Baxter's base frame and the
267  camera frame using singular value decomposition.
268 
269  :param n: The number of absolute end effector poses to use.
270  :param tto: The homogeneous transform between TCP and the origin of
271  the calibration pattern.
272  :return: The homogeneous transform between Baxter's base frame and
273  the camera frame.
274  """
275  bto = list()
276  cto = list()
277 
278  self.logger.info("Record %d absolute point cloud pairs." % n)
279  pattern = self._pattern.T[:, np.newaxis, :]
280  hom_pattern = np.concatenate([self._pattern,
281  np.ones((1, self._pattern.shape[1]))],
282  axis=0)
283  while len(bto) < n and not rospy.is_shutdown():
284  if self._move():
285  rospy.sleep(1.0)
286  bttn = self._robot.hom_gripper_to_robot(arm=self._arm)
287  color, _, _ = self._kinect.collect_data(color=True)
288 
289  self._pub_vis.publish(img_to_imgmsg(img=color))
290  patternfound, centers = cv2.findCirclesGridDefault(
291  image=color, patternSize=self._patternsize,
292  flags=cv2.CALIB_CB_ASYMMETRIC_GRID)
293  if patternfound == 0:
294  self.logger.debug("No pattern found!")
295  continue
296 
297  rot, trans, _ = cv2.solvePnPRansac(
298  objectPoints=pattern, imagePoints=centers,
299  cameraMatrix=self._kinect.color.camera_matrix,
300  distCoeffs=self._kinect.color.distortion_coeff,
301  flags=cv2.CV_ITERATIVE)
302  rot, _ = cv2.Rodrigues(rot)
303  cton = np.eye(4)
304  cton[:-1, :-1] = rot
305  cton[:-1, -1] = np.squeeze(trans)
306 
307  fname = os.path.join(self._sink, "2_tto_%d" % (len(bto) + 1))
308  cv2.imwrite(fname + ".jpg", color)
309  cv2.drawChessboardCorners(image=color,
310  patternSize=self._patternsize,
311  corners=centers,
312  patternWasFound=patternfound)
313  self._pub_vis.publish(img_to_imgmsg(img=color))
314  cv2.imwrite(fname + "_det.jpg", color)
315 
316  bto.append(np.dot(np.dot(bttn, tto), hom_pattern))
317  fname = os.path.join(self._sink, "2_bto_%d.npz" % len(bto))
318  np.savez(fname, bttn=bto[-1])
319  cto.append(np.dot(cton, hom_pattern))
320  fname = os.path.join(self._sink, "2_cto_%d.npz" % len(cto))
321  np.savez(fname, cton=cto[-1])
322 
323  self.logger.info("Compute affine transformation from 3D point "
324  "correspondences.")
325  # see http://stackoverflow.com/questions/15963960/opencv-2-4-estimateaffine3d-in-python
326  # data of shape (Nx3)
327  cto = np.array(cto).reshape((4, -1)).T[:, :-1]
328  bto = np.array(bto).reshape((4, -1)).T[:, :-1]
329  cto_mean = cto.mean(axis=0)
330  bto_mean = bto.mean(axis=0)
331  # compute covariance
332  cov = np.dot((cto - cto_mean).T, bto - bto_mean)
333  u, s, v = np.linalg.svd(cov)
334  rot = np.dot(u, v.T)
335  if np.linalg.det(rot) < 0:
336  v[:, 2] = -v[:, 2]
337  rot = np.dot(u, v.T)
338  trans = bto_mean - np.dot(rot, cto_mean)
339  trafo = np.eye(4)
340  trafo[:-1, :-1] = rot
341  trafo[:-1, -1] = trans
342  fname = os.path.join(self._sink, "2_btc.npz")
343  np.savez(fname, btc=trafo)
344  return trafo
