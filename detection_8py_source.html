91  def detect(self, image):
92  """Feed forward the given image through the previously loaded network.
93  Return scores and bounding boxes for all abject proposals and classes.
94 
95  :param image: An image (numpy array) of shape (height, width, 3).
96  :return: A tuple of two numpy arrays, the n_proposals x n_classes
97  scores and the corresponding n_proposals x 4*n_classes bounding
98  boxes, where each bounding box is defined as <xul, yul, xlr, ylr>.
99  """
100  if self._net is None:
101  raise RuntimeError("No loaded network found! "
102  "Did you run init_model()?")
103  if len(image.shape) != 3 and image.shape[2] != 3:
104  raise ValueError("Image must be a three channel color image "
105  "with shape (h, w, 3)!")
106  start = time.time()
107  scores, boxes = im_detect(self._net, image)
108  self._logger.debug('Detection took {:.3f}s for {:d} object proposals'.format(
109  time.time() - start, boxes.shape[0])
110  )
111  return scores, boxes

113  def detect_object(self, image, object_id, threshold=0.5):
114  """Feed forward the given image through the previously loaded network.
115  Return the bounding box with the highest score for the requested
116  object class.
117 
118  :param image: An image (numpy array) of shape (height, width, 3).
119  :param object_id: One object identifier string contained in the list
120  of objects.
121  :param threshold: The threshold (0, 1) on the score for a detection
122  to be considered as valid.
123  :return: A dictionary containing the detection with
124  'id': The object identifier.
125  'score: The score of the detection (scalar).
126  'box': The bounding box of the detection; a (4,) numpy array.
127  """
128  if object_id not in self._classes:
129  raise KeyError("Object {} is not contained in the defined "
130  "set of objects!".format(object_id))
131  scores, boxes = self.detect(image=image)
132 
133  # Find scores for requested object class
134  cls_idx = self._classes.index(object_id)
135  cls_scores = scores[:, cls_idx]
136  cls_boxes = boxes[:, 4*cls_idx:4*(cls_idx + 1)]
137 
138  best_idx = np.argmax(cls_scores)
139  best_score = cls_scores[best_idx]
140  best_box = cls_boxes[best_idx]
141  self._logger.debug('Best score for {} is {:.3f} {} {:.3f}'.format(
142  object_id,
143  best_score,
144  '>=' if best_score > threshold else '<',
145  threshold)
146  )
147  if best_score > threshold:
148  return {'id': object_id, 'score': best_score, 'box': best_box}
149  return {'id': object_id, 'score': best_score, 'box': None}

151  def detect_best(self, image, threshold=0.5):
152  """Feed forward the given image through the previously loaded network.
153  Return the bounding box with the highest score amongst all classes.
154 
155  :param image: An image (numpy array) of shape (height, width, 3).
156  :param threshold: The threshold (0, 1) on the score for a detection
157  to be considered as valid.
158  :return: A dictionary containing the detection with
159  'id': The object identifier.
160  'score: The score of the detection (scalar).
161  'box': The bounding box of the detection; a (4,) numpy array.
162  """
163  scores, boxes = self.detect(image=image)
164 
165  # find best score among all classes (except background)
166  best_proposal, best_class = np.unravel_index(scores[:, 1:].argmax(),
167  scores[:, 1:].shape)
168  best_class += 1 # compensate for background
169  best_score = scores[best_proposal, best_class]
170  best_box = boxes[best_proposal, 4*best_class:4*(best_class + 1)]
171  best_object = self._classes[best_class]
172 
173  self._logger.debug('Best score for {} is {:.3f} {} {:.3f}'.format(
174  best_object,
175  best_score,
176  '>=' if best_score > threshold else '<',
177  threshold)
178  )
179  if best_score > threshold:
180  return {'id': best_object, 'score': best_score, 'box': best_box}
181  return {'id': best_object, 'score': best_score, 'box': None}
