33 def wait_for_instruction():

37  rospy.wait_for_service('demo_instruction', timeout=5.0)

40  "Did you run 'rosrun baxter_pick_and_place instruct.py'?")

42  service = rospy.ServiceProxy('demo_instruction', Trigger)
43  instr = service()
44  return instr.message
45  except rospy.ServiceException as e:
46  logger.error("service call failed: {}".format(e))
