29 class ServoingDistance(Servoing):
30  def estimate_distance(self, object_id, rroi, arm):
31  """Estimate the distance to the object from
32  - the measured pose of the end effector and
33  - the measured height of the table top in robot coordinates
34  by computing
35  distance = height_gripper - height_table.
36 
37  :param object_id: The object identifier of the object to estimate the
38  distance to.
39  :param rroi: The rotated rectangle enclosing the segmented object,
40  given by ((cx, cy), (w, h), alpha).
41  :param arm: The arm <'left', 'right'> to control.
42  :return: The approximate distance from the gripper to the object.
43  """
44  return self._robot.endpoint_pose(arm=arm)[2] - self._robot.z_table
45 
46  def correct_height(self, arm):
47  """Make sure the gripper height is appropriate before attempting to
48  grasp the object.
49 
50  :param arm: The arm <'left', 'right'> to control.
51  :return: A boolean success value.
52  """
53  pose = self._robot.endpoint_pose(arm=arm)
54  pose[2] = self._robot.z_table + 0.01
55  try:
56  cfg = self._robot.ik(arm=arm, pose=pose)
57  self._robot.move_to_config(config=cfg)
58  except ValueError as e:
59  self._logger.error(e)
60  return False
61  return True
