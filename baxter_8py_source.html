49  def __init__(self, sim=False):
50  """Hardware abstraction of the Baxter robot using the BaxterSDK
51  interface.
52 
53  :param sim: Whether in Gazebo (True) or on real Baxter (False).
54  """
55  name = 'main.baxter'
56  self._logger = logging.getLogger(name)
57  self._arms = ['left', 'right']
58  self._limbs = {a: baxter_interface.Limb(a)
59  for a in self._arms}
60  self._grippers = {a: baxter_interface.Gripper(a)
61  for a in self._arms}

63  self._grippers_pars['moving_force'] = 40.0
64  self._grippers_pars['holding_force'] = 30.0
65  self._sensors = {a: baxter_interface.analog_io.AnalogIO('%s_hand_range' % a)
66  for a in self._arms}
67  # Cameras on the Baxter robot are tricky. Due to limited bandwidth
68  # only two cameras can be operating at a time.
69  # http://sdk.rethinkrobotics.com/wiki/Camera_Control_Tool
70  # Default behavior on Baxter startup is for both of the hand cameras
71  # to be in operation at a resolution of 320x200 at a frame rate of
72  # 25 fps. We get their CameraControllers using the Baxter SDK ...
73  self.cameras_d = {a: baxter_interface.CameraController('%s_hand_camera' % a, sim=sim)
74  for a in self._arms}
75  # ... and set their resolution to 1280x800 @ 14 fps.
76  for arm in self._arms:
77  self.cameras_d[arm].resolution = (1280, 800)
78  self.cameras_d[arm].fps = 14.0
79  self.cameras_d[arm].exposure = settings.baxter_cam_exposure
80  # We don't need the CameraControllers any more. Our own module will
81  # do the remaining camera handling for us.
82  self.cameras = {a: Camera(topic='/cameras/{}_hand_camera/image'.format(a),
83  prefix=name)
84  for a in self._arms}

229  def ik(self, arm, pose=None):
230  """Solve inverse kinematics for one limb at given pose.
231 
232  :param arm: The arm <'left', 'right'> to control.
233  :param pose: The pose to stamp. One of
234  - None, in which case the current set of joint angles is returned,
235  - a ROS Pose,
236  - a list of length 6 [x, y, z, roll, pitch, yaw] or
237  - a list of length 7 [x, y, z, qx, qy, qz, qw].
238  :return:
239  """
240  if pose is None:
241  return self._limbs[arm].joint_angles()
242 
243  pq = self._stamp_pose(pose, target_frame="base")
244  node = "ExternalTools/" + arm + "/PositionKinematicsNode/IKService"
245  ik_service = rospy.ServiceProxy(node, SolvePositionIK)
246  ik_request = SolvePositionIKRequest()
247  ik_request.pose_stamp.append(pq)
248  try:
249  rospy.wait_for_service(node, 5.0)
250  ik_response = ik_service(ik_request)
251  except (rospy.ServiceException, rospy.ROSException), error_message:
252  self._logger.error("Service request failed: %r" % (error_message,))
253  raise
254 
255  if ik_response.isValid[0]:
256  # convert response to joint position control dictionary
257  return dict(zip(ik_response.joints[0].name,
258  ik_response.joints[0].position))
259  else:
260  pose_str = np.array_str(np.array(pose), precision=3,
261  suppress_small=True)
262  s = "No valid configuration found for " \
263  "pose {} with {} arm!".format(pose_str, arm)
264  self._logger.debug(s)
265  raise ValueError(s)

267  def ik_either_limb(self, pose):
268  """Attempt to solve the inverse kinematics for a given pose with
269  either arm. If no solution is found, raise an exception
270 
271  :param pose: The pose to stamp. One of
272  - None, in which case the current set of joint angles is returned,
273  - a ROS Pose,
274  - a list of length 6 [x, y, z, roll, pitch, yaw] or
275  - a list of length 7 [x, y, z, qx, qy, qz, qw].
276  :return: tuple of string and dict:
277  - the arm <'left', 'right'> the solution was found for
278  - a dictionary of joint name keys to joint angles.
279  :raise ValueError: if no valid configuration was found for either arm.
280  """
281  arm = 'left'
282  try:
283  cfg = self.ik(arm, pose)
284  except ValueError:
285  # no valid configuration found for left arm
286  arm = 'right'
287  try:
288  cfg = self.ik(arm, pose)
289  except ValueError:
290  # no valid configuration found for right arm
291  s = "No valid configuration found for pose {} with either arm!".format(pose)
292  self._logger.warning(s)
293  raise ValueError(s)
294  return arm, cfg

296  def control(self, trajectory):
297  """Control one limb using position, velocity or torque control.
298 
299  :param trajectory: A generator MotionPlanner instance.
300  :return:
301  """
302  if not isinstance(trajectory, MotionPlanner):
303  raise TypeError("'trajectory' must be a MotionPlanner instance!")
304  if trajectory.controller_type == 'position':
305  for q in trajectory:
306  arm = q.keys()[0].split('_')[0]
307  self._limbs[arm].move_to_joint_positions(q)
308  elif trajectory.controller_type == 'velocity':
309  raise NotImplementedError("Need to implement velocity control!")
310  # for v in trajectory:
311  # self._limbs[arm].set_joint_velocities(v)
312  elif trajectory.controller_type == 'torque':
313  raise NotImplementedError("Need to implement torque control!")
314  # for t in trajectory:
315  # self._limbs[arm].set_joint_torques(t)
316  else:
317  raise KeyError("No such control mode: '{}'!".format(trajectory.controller_type))

389  def select_gripper_for_object(self, object_id):
390  """Select the most suitable gripper for the grasp width corresponding
391  to a given object identifier.
392 
393  :param object_id: The object identifier.
394  :return: The most appropriate limb <'left', 'right'> to grasp the
395  requested object.
396  """
397  dist = 0.0
398  min_width = 0.0
399  arm = None
400 
401  size = settings.object_size_meters[object_id]
402  for a in self._arms:
403  gr = self._gripper_ranges_meters()[settings.gripper_settings[a]]
404  if gr[0] <= size < gr[1]:
405  dists = [abs(x - size) for x in gr]
406  min_dist = min(dists)
407  if min_dist > 0.0:
408  self._logger.debug("{} gripper is suitable to grasp object "
409  "(offset={:.3f} m).".format(
410  a.capitalize(), min_dist)
411  )
412  if min_dist > dist:
413  dist = min_dist
414  if gr[0] > min_width:
415  min_width = gr[0]
416  arm = a
417  elif min_dist == dist:
418  # select wider gripper configuration
419  if gr[0] > min_width:
420  min_width = gr[0]
421  arm = a
422  if arm is None:
423  msg = ("No suitable gripper for object {} ({:.3f} m) installed! "
424  "Check your gripper settings. Currently installed ranges "
425  "are {:.3f}--{:.3f} m and {:.3f}--{:.3f} m.".format(
426  object_id,
427  settings.object_size_meters[object_id],
428  self._gripper_ranges_meters()[settings.gripper_settings['left']][0],
429  self._gripper_ranges_meters()[settings.gripper_settings['left']][1],
430  self._gripper_ranges_meters()[settings.gripper_settings['right']][0],
431  self._gripper_ranges_meters()[settings.gripper_settings['right']][1])
432  )
433  self._logger.error(msg)
434  raise ValueError(msg)
435  self._logger.info("Selected {} limb to grasp object.".format(arm))
436  return arm

491  def hom_camera_to_robot(self, arm):
492  """Get the homogeneous transformation matrix {}^R\mat{T}_{C} relating
493  camera coordinates to robot coordinates.
494 
495  :param arm: The arm <'left', 'right'> to control.
496  :return: The homogeneous transformation matrix (a 4x4 numpy array).
497  """
498  hom_grip_in_rob = self.hom_gripper_to_robot(arm=arm)
499  hom_cam_in_grip = np.eye(4)
500  hom_cam_in_grip[:-1, :-1] = np.array([[0, 1, 0], [-1, 0, 0], [0, 0, 1]])
501  hom_cam_in_grip[:-1, -1] = self.cam_offset
502  hom_cam_in_rob = np.dot(hom_grip_in_rob, hom_cam_in_grip)
503  return hom_cam_in_rob

514  def estimate_object_position(self, arm, center):
515  """Compute an estimate for the 3D position of an object lying on a
516  table with known height.
517  Note: This method only works if the gripper is restricted to be
518  oriented perpendicular to the table top.
519 
520  :param arm: The arm <'left', 'right'> to control.
521  :param center: The pixel coordinates to project to robot coordinates.
522  :return: The estimated object position as a list of length 3 [x, y, z].
523  """
524  distance = self.camera_pose(arm=arm)[2] - self.z_table
525  cam_coord = self.cameras[arm].projection_pixel_to_camera(pixel=center,
526  z=distance)
527  hom_coord = np.asarray(cam_coord + [1])
528  rob_coord = np.dot(self.hom_camera_to_robot(arm=arm), hom_coord)
529  rob_coord /= rob_coord[-1]
530  delta = abs(abs(rob_coord[2]) - abs(self.z_table))
531  if delta > 1e-3:
532  self._logger.warning("Estimated and measured z coordinate of the object "
533  "(table) deviate by {} > 0.001 m!".format(delta))
534  return [rob_coord[0], rob_coord[1], self.z_table]
