40 def textbox(image, text, org, font_face, font_scale, thickness, color, color_box):
41  """Draw a filled box with text placed on top of it.
42 
43  :param image: The image to draw on.
44  :param text: The text to put there.
45  :param org: The origin of the text (bottom left corner) (x, y).
46  :param font_face: The font to use.
47  :param font_scale: The scale for the font.
48  :param thickness: The thickness of the font.
49  :param color: The color of the font (BGR).
50  :param color_box: The color of the box (BGR).
51  :return:
52  """
53  (w, h), _ = cv2.getTextSize(text=text,
54  fontFace=font_face, fontScale=font_scale,
55  thickness=thickness)
56  ox, oy = [int(o) for o in org]
57  cv2.rectangle(image, pt1=(ox - 2, oy + 2), pt2=(ox + 2 + w, oy - 2 - h),
58  color=color_box, thickness=cv2.cv.CV_FILLED)
59  cv2.putText(image, text=text, org=(ox, oy),
60  fontFace=font_face, fontScale=font_scale,
61  thickness=thickness, color=color)

64 def draw_detection(image, detections):
65  """Draw all given detections onto the given image and label them with
66  their object identifier and score.
67  Note: Modifies the passed image!
68 
69  :param image: An image (numpy array) of shape (height, width, 3).
70  :param detections: A (list of) dictionary(ies) of detection containing
71  'id': The object identifier.
72  'score: The score of the detection (scalar).
73  'box': The bounding box of the detection; a (4,) numpy array.
74  ['mask': The segmentation of the detection; a (height, width,)
75  numpy array.]
76  :return:
77  """
78  if not isinstance(detections, list):
79  detections = [detections]
80  for detection in detections:
81  if 'mask' in detection and detection['mask'] is not None:
82  image[detection['mask'] == 255] = yellow
83  if detection['box'] is not None:
84  oid, s, b = [detection['id'], detection['score'], detection['box']]
85  cv2.rectangle(image, pt1=(b[0], b[1]), pt2=(b[2], b[3]),
86  color=red, thickness=2)
87  textbox(image, text='%s %.3f' % (oid, s), org=(b[0] + 3, b[3] - 3),
88  font_face=cv2.FONT_HERSHEY_SIMPLEX,
89  font_scale=0.5, thickness=2, color=black, color_box=white)

92 def mask_to_rroi(mask):
93  """Compute the minimum enclosing rectangle (possibly rotated) for a given
94  pre-computed segmentation of an object.
95 
96  :param mask: The binary mask image to compute the minimum enclosing
97  rectangle (possibly rotated) for.
98  :return: The minimum enclosing rectangle, defined by
99  <(cx, cy), (w, h), alpha>, where w and h are the width and height of
100  the rectangle centered around cx and cy and rotated by alpha degrees.
101  Note: Alpha is defined *clockwise* starting from the positive x axis
102  and is forced to point along the shorter dimension of the rotated
103  rectangle.
104  """
105  contours, _ = cv2.findContours(image=mask, mode=cv2.RETR_LIST,
106  method=cv2.CHAIN_APPROX_SIMPLE)
107  if len(contours) > 1:
108  raise ValueError("Expected to find exactly one contour!")
109  center, (w, h), alpha = cv2.minAreaRect(contours[0])
110  # make sure alpha always points along the shorter dimension of the mask
111  # (this is the dimension along which we want to grasp the object)
112  if w > h:
113  temp = h
114  h = w
115  w = temp
116  alpha = (alpha + 90.0) % 360.0
117  elif h == w:
118  alpha = min(alpha, (alpha + 90.0) % 360.0)
119  else:
120  # h > w, no need to do anything
121  pass
122  return center, (w, h), alpha

125 def draw_rroi(image, rroi):
126  """Draw the given rotated rectangle onto the given image.
127  Note: Modifies the passed image!
128 
129  :param image: An image (numpy array) of shape (height, width, 3).
130  :param rroi: The rotated rectangle ((cx, cy), (w, h), angle) to draw.
131  :return:
132  """
133  h, w = image.shape[:2]
134  # visualize image center (=target)
135  cv2.circle(image, center=(w//2, h//2), radius=4, color=blue, thickness=2)
136  # visualize rroi center and orientation
137  center = tuple(np.round(rroi[0]).astype(np.uint32))
138  cv2.circle(image, center=center, radius=4, color=green, thickness=2)
139  angle = np.deg2rad(rroi[2])
140  line_length = 20
141  pt2 = [a + b
142  for a, b in zip(rroi[0], (line_length*np.cos(angle),
143  line_length*np.sin(angle)))]
144  pt2 = tuple(np.round(pt2).astype(np.uint32))
145  cv2.line(image, pt1=center, pt2=pt2, color=green, thickness=1)
146  # draw rotated rectangle
147  box = np.int0(cv2.cv.BoxPoints(rroi))
148  cv2.drawContours(image, contours=[box], contourIdx=0, color=green,
149  thickness=2)

152 def color_difference(image_1, image_2):
153  """Compute the pixel-wise delta E color difference for two RGB images.
154  Note: We compare RGB images, which means that the result is not as
155  accurate as if we converted the images to LAB color space before
156  computing the delta E difference. This way the result is nicely
157  interpretable, though.
158  See
159  http://jeffkreeftmeijer.com/2011/comparing-images-and-creating-image-diffs/
160  and
161  https://www.cis.rit.edu/~cnspci/media/software/deltae.py
162  for more information.
163 
164  :param image_1: A color image ((h, w, 3) uint8 numpy array).
165  :param image_2: A color image ((h, w, 3) uint8 numpy array).
166  :return: The delta E color difference as a (h, w) float numpy array with
167  values in the range [0, 1] and
168  the the corresponding gray scale image.
169  """
170  assert image_1.shape == image_2.shape
171  # Scale both color images to unit range
172  image_1, image_2 = [img.astype(np.float32)/255.
173  for img in [image_1, image_2]]
174  # compute the delta E color difference per pixel
175  delta_e = np.sqrt(np.sum((image_2 - image_1)**2, axis=-1)/3.)
176  return delta_e, np.asarray(255.*delta_e, dtype=np.uint8)
