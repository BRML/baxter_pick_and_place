36 def angle_axis_rotation_matrix(angle, axis):
37  # R = d d.T + cos(a) (I - d d.T) + sin(a) skew(d)
38  mag = np.linalg.norm(axis, ord=2)
39  if mag == 0.0:
40  return np.eye(3)
41  d = axis/mag
42  ddt = np.outer(d, d)
43  rot = ddt + np.cos(angle)*(np.eye(3) - ddt) + np.sin(angle)*skew(d)
44  return rot

122  # solve for rotation
123  n = len(a)
124  na = np.zeros(3*n)
125  nb = np.zeros(3*n)

128  # treat imaginary part of quaternion as angle-axis representation
129  q_a = 2*quaternion_from_rotation_matrix(rot=trafo_a[:-1, :-1])[:-1]
130  q_b = 2*quaternion_from_rotation_matrix(rot=trafo_b[:-1, :-1])[:-1]

132  na[3*i:3*i+3] = q_a
133  nb[3*i:3*i+3] = q_b
134  snapnb[3*i:3*i+3, :] = skew(v=q_a + q_b)

136  if rank < 3:

139  m_, float(res) if len(res) > 0 else 'NaN')
