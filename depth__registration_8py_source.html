26 # DISCLAIMER:
27 # When implementing a C++ node using the kinect2_registration library from
28 # the iai_kinect2 package we found that the depth image is cropped before
29 # being scaled and the field of view of the resulting color and depth images
30 # are quite different, such that they cannot easily be superimposed. We thus
31 # resort to a very naive workaround by scaling and shifting the depth image
32 # by constant factors to match the color image reasonable well.
33 # This is feasible since all we need is a rough estimate for the depth, the
34 # visual servoing will take care of the rest.
35 # It would be much nicer to use the iai_kinect2 or WindowsSDK libraries,
36 # though.

43 def register_depth(img_depth, size_color):
44  """Compute a depth image that matches the color image.
45  Note: This method implements a naive workaround to 'proper' depth image
46  registration, as for instance implemented in the iai_kinect2 package.
47 
48  :param img_depth: A depth image.
49  :param size_color: The size of the color image.
50  :return: A depth image matching the color image.
51  """
52  if size_color == (540, 960):
53  # magic factors for matching the depth image to the 960x540 color image
54  scale_factor = 1.45
55  x_shift = 120
56  y_shift = -38
57  elif size_color == (1080, 1920):
58  # magic factors for matching the depth image to the 1920x1080 color image
59  # TODO: modify magic factors for full HD resolution
60  scale_factor = 1.45*2
61  x_shift = 120*2
62  y_shift = -38*2
63  else:
64  raise ValueError("Not defined for size {}!".format(size_color))
65  dsize = tuple([int(x * scale_factor)
66  for x in img_depth.shape[:2]][::-1])
67  img_depth = cv2.resize(img_depth, dsize, interpolation=cv2.INTER_CUBIC)
68  output = np.zeros(size_color, dtype=img_depth.dtype)
69  h0, w0 = output.shape[:2]
70  h2, w2 = img_depth.shape[:2]
71  # where to put image
72  x0 = min(max(x_shift, 0), w0)
73  y0 = min(max(y_shift, 0), h0)
74  x1 = min(x_shift + w2, w0)
75  y1 = min(y_shift + h2, h0)
76  # what to put
77  x2 = min(-x_shift if x_shift < 0 else 0, w2)
78  y2 = min(-y_shift if y_shift < 0 else 0, h2)
79  x3 = min(x2 + x1 - x0, w2)
80  y3 = min(y2 + y1 - y0, h2)
81  output[y0:y1, x0:x1] = img_depth[y2:y3, x2:x3]
82  return output

85 def get_depth(img_depth, size_color, pixel_color):
86  """Compute the depth at a given pixel in the RGB image from a corresponding
87  depth image.
88 
89  :param img_depth: A depth image.
90  :param size_color: The size of the color image.
91  :param pixel_color: The requested pixel in the color image.
92  :return: The depth at the requested pixel.
93  """
94  if size_color == (540, 960):
95  # magic factors for matching the depth image to the 960x540 color image
96  scale_factor = 1.45
97  x_shift = 120
98  y_shift = -38
99  elif size_color == (1080, 1920):
100  # magic factors for matching the depth image to the 1920x1080 color image
101  # TODO: modify magic factors for full HD resolution
102  scale_factor = 1.45*2
103  x_shift = 120*2
104  y_shift = -38*2
105  else:
106  raise ValueError("Not defined for size {}!".format(size_color))
107  cx, cy = pixel_color
108  scaled_x = int((cx - x_shift)/scale_factor)
109  scaled_y = int((cy - y_shift)/scale_factor)
110  return img_depth[scaled_y, scaled_x]/1000.0

113 def blend(img_color, img_depth):
114  """Blend the color and registered depth image on top of each other.
115 
116  :param img_color: A color image.
117  :param img_depth: A corresponding depth image.
118  :return: The registered depth image overlaid on top of the color image.
119  """
120  img_reg = register_depth(img_depth, img_color.shape[:2])
121  img_reg = cv2.cvtColor(cv2.convertScaleAbs(img_reg), cv2.COLOR_GRAY2BGR)
122  alpha = 0.3
123  beta = 1.0 - alpha
124  gamma = 0.0
125  return cv2.addWeighted(img_color, alpha, img_reg, beta, gamma)
