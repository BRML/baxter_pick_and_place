99  def _receive_size(self):
100  """Receive the number of bytes needed to read from the data stream.
101 
102  :return: The _image_size of the data to read (as an int).
103  """
104  data = self._socket.recv(4)
105  try:
106  size = struct.unpack('<i', data)[0] # we receive an int value
107  except ValueError:
108  size = -1
109  finally:
110  # Sending ACK that we received the _image_size
111  self._socket.sendall("OK\n")
112  return size

114  def _receive_data(self, n_bytes):
115  """Receive a given number of bytes from the data stream provided by
116  the ELTE Kinect Windows tool.
117 
118  :param n_bytes: The number of bytes to read from the data stream.
119  :return: The received data.
120  :raise ValueError: If the requested number of bytes was not received.
121  """
122  # Reading the socket stream to get every packet
123  data = self._socket.recv(n_bytes)
124  while len(data) < n_bytes:
125  data += self._socket.recv(n_bytes - len(data))
126  self._logger.debug("Received {}/{} bytes.".format(len(data), n_bytes))
127  if len(data) != n_bytes:
128  msg = "Received {} but should have received {} bytes!".format(
129  len(data), n_bytes)
130  self._logger.error(msg)
131  raise ValueError(msg)
132  # Sending ACK that we received the data
133  self._socket.sendall("OK2\n")
134  return data

136  def _receive_color(self):
137  """Receive a color image from the ELTE Kinect Windows tool and decode
138  it as a uint8, three-channel numpy array of _image_size height x width.
139 
140  :return: A (h, w, 3) numpy array holding the color image.
141  """
142  start = time.time()
143  # Reading the _image_size of the data stream we want to read
144  n_bytes = self._receive_size()
145  if n_bytes == -1:
146  self._logger.warning("Failed to receive color image data!")
147  return None
148  self._logger.debug("Need to receive {} bytes.".format(n_bytes))
149 
150  # Reading the data from the socket stream
151  try:
152  msg = self._receive_data(n_bytes=n_bytes)
153  except ValueError:
154  return None
155 
156  # Convert received data into a numpy array
157  try:
158  barray = np.fromstring(msg, np.uint8)
159  except ValueError:
160  self._logger.warning("Error when converting raw data to color image!")
161  return None
162  img = cv2.imdecode(barray, cv2.IMREAD_COLOR)
163  self._logger.debug("Received a {} {} color image (min={}, max={}) "
164  "in {:.3f} s.".format(img.shape, img.dtype,
165  img.min(), img.max(),
166  time.time() - start))
167  return img

207  def _receive_skeleton_data(self, n_bytes):
208  """Receive skeleton data from the ELTE Kinect Windows tool.
209 
210  :param n_bytes: The number of bytes to read from the data stream.
211  :return: A tuple containing the number of bodies (skeletons) in the

227  except ValueError:

231  def _receive_skeleton(self):
232  """Receive skeleton data from the ELTE Kinect Windows tool and decode
233  if as a list of length number of bodies (skeletons), where each
234  skeleton is defined by three lists holding 13 joint coordinates each,
235  where the first list holds the camera coordinates, the second list
236  holds color space (pixel) coordinates and the third list holds depth
237  space (pixel) coordinates.
238 
239  :return: A list holding n skeletons defined by three lists of joint
240  coordinates in camera, color and depth space.
241  """
242  start = time.time()
243  # Reading the _image_size of the data stream we want to read
244  n_bytes = self._receive_size()
245  if n_bytes == -1:
246  msg = "Failed to receive skeleton data!"
247  self._logger.warning(msg)
248  return list()
249  self._logger.debug("Need to receive {} bytes.".format(n_bytes))
250 
251  # Reading the data from the socket stream
252  try:
253  n_bodies, msg = self._receive_skeleton_data(n_bytes=n_bytes)
254  except ValueError:
255  return list()
256 
257  # Convert received data into a list
258  try:
259  barray = np.fromstring(msg, np.float32)
260  except ValueError:
261  self._logger.warning("Error when converting raw data to skeleton list!")
262  return list()
263  bodies = list()
264  for body in range(n_bodies):
265  cam_space_points = list()
266  color_space_points = list()
267  depth_space_points = list()
268  for i in range(body*self.joint_type_count*7,
269  (body + 1)*self.joint_type_count*7, 7):
270  cam_space_points.append((barray[i], barray[i + 1], barray[i + 2]))
271  color_space_points.append((barray[i + 3], barray[i + 4]))
272  depth_space_points.append((barray[i + 5], barray[i + 6]))
273  bodies.append((cam_space_points, color_space_points, depth_space_points))
274  self._logger.debug("Received skeleton data for {} bod{} in {:.3f} s.".format(
275  len(bodies), 'y' if len(bodies) == 1 else 'ies',
276  time.time() - start))
277  return bodies

279  def collect_data(self, color=False, depth=False, skeleton=False):
280  """Collect the latest color, depth and skeleton data from the Kinect V2
281  sensor.
282  Note: If the Kinect is connected to a Ubuntu machine, use the native
283  ROS interface using the libfreenect2 and iai_kinect2 libraries. If the
284  Kinect is connected to a Windows machine and runs the ELTE Kinect
285  Windows tool server, communicate via a TCP/IP socket connection.
286 
287  :param color: Whether to retrieve the latest color image.
288  :param depth: Whether to retrieve the latest depth image.
289  :param skeleton: Whether to retrieve the latest skeleton data.
290  :return: A triple (color image, depth image, skeletons).
291  """
292  img_color = None
293  img_depth = None
294  data_skeleton = list()
295  if self._native_ros:
296  # Kinect is connected to the Ubuntu machine, communicate via ROS
297  if color:
298  img_color = self.color.collect_image()
299  if depth:
300  img_depth = self.depth.collect_image()
301  if skeleton:
302  # libfreenect2 does not provide skeleton data
303  pass
304  else:
305  # Kinect is connected to a Windows machine, communicate via
306  # TCP/IP socket connection
307  if not self._host:
308  msg = "No host name for ELTE Kinect Windows tool provided!"
309  self._logger.error(msg)
310  raise ValueError(msg)
311  # Create a TCP/IP socket
312  self._socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
313  # Connect the socket to the host and port where the server listens
314  server = self._host, 9999
315  self._logger.debug('Connect to {} on port {}.'.format(server[0].upper(),
316  server[1]))
317  try:
318  self._socket.connect(server)
319  except socket.gaierror:
320  raise RuntimeError("Failed to connect to {}!".format(server[0].upper()) +
321  " Check your network connection.")
322  try:
323  msg = '{}{}{}\n'.format(*[1 if x else 0
324  for x in [color, depth, skeleton]])
325  self._socket.sendall(msg)
326  if color:
327  img_color = self._receive_color()
328  if depth:
329  img_depth = self._receive_depth()
330  if skeleton:
331  data_skeleton = self._receive_skeleton()
332  except socket.error as e:
333  self._logger.error(str(e))
334  finally:
335  self._logger.debug('Close socket.')
336  self._socket.close()
337  self._socket = None
338  # transmitting depth images sometimes fails
339  if depth and img_depth is None:
340  return self.collect_data(color=color, depth=depth, skeleton=skeleton)
341  return img_color, img_depth, data_skeleton

343  def estimate_hands_positions(self):
344  """Extract the estimate for the approximate hand position from the
345  skeleton data obtained from the Kinect.
346 
347  :return: One of
348  - A dictionary holding the hand coordinate triplets (camera,
349  color and depth space coordinates) for the left and right hands.
350  - None if no skeleton estimate is computed by the Kinect.
351  """
352  color, _, skeletons = self.collect_data(color=True, skeleton=True)
353  if len(skeletons) != 1:
354  raise ValueError("Need to track exactly one person!")
355  skeleton = skeletons[0]
356  if skeleton is None:
357  estimate = None
358  else:
359  estimate = dict()
360  for arm, idx in zip(['left', 'right'],
361  [self.joint_type_hand_left,
362  self.joint_type_hand_right]):
363  coord, px_color, px_depth = [a[idx] for a in skeleton]
364  # The ELTE KinectOverNetwork tool scales the color image by a
365  # factor of 1/2 (to 960x540). We thus adapt the estimated pixel
366  # coordinates accordingly.
367  px_color = tuple(x/2.0 if not self._native_ros else x
368  for x in px_color)
369  pos = tuple(np.dot(self.trafo, list(coord) + [1])[:-1])
370  estimate[arm] = (pos, px_color, px_depth)
371  # visualize estimate
372  ctr = tuple(int(x) for x in px_color)
373  cv2.circle(color, center=ctr, radius=5, color=[255, 0, 0],
374  thickness=3)
375  self._pub_vis.publish(img_to_imgmsg(img=color))
376  return estimate

395  def estimate_object_position(self, img_color, bbox, img_depth):
396  """Estimate the approximate position of an object in 3d from a Kinect
397  color and corresponding depth image, as well as the bounding box of
398  the object detected in the color image.
399 
400  :param img_color: A color image.
401  :param bbox: The bounding box of the object we are interested in in
402  the color image.
403  :param img_depth: A depth image corresponding to the color image.
404  :return: A list [x, y, z] representing the approximate object position
405  in camera coordinates or None if the object was not detected.
406  """
407  if bbox is None:
408  return None
409 
410  if len(bbox) == 3:
411  # smallest enclosing rectangle
412  px, py = bbox[0]
413  elif len(bbox) == 4:
414  # bounding box
415  px, py = bbox[2] - bbox[0], bbox[3] - bbox[1]
416  else:
417  raise ValueError("Expected rroi or bounding box, got {}!".format(bbox))
418  z_3d = get_depth(img_depth, img_color.shape[:2], (px, py))
419  pos_cam = self.color.projection_pixel_to_camera(pixel=(px, py), z=z_3d)
420  pos_rob = np.dot(self.trafo, pos_cam + [1])[:-1]
421  return list(pos_rob)
