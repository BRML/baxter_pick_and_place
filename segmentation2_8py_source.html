44  def _morphology(self, image):
45  """Using a number of image morphology operators to find the single
46  connected region in the image.
47  Note: This method will fail as soon as there are multiple objects in
48  the given image patch (i.e., objects on the table ly to close to
49  each other.
50 
51  :param image: A binary input image of shape (height, width).
52  :return: A triple containing
53  - a list of contours,
54  - the bounding box, or None and
55  - the segmentation mask, or None.
56  """
57  kernel = cv2.getStructuringElement(shape=cv2.MORPH_ELLIPSE, ksize=(3, 3))
58  opening = cv2.morphologyEx(image, cv2.MORPH_OPEN, kernel)
59  contour, _ = cv2.findContours(opening, cv2.RETR_CCOMP, cv2.CHAIN_APPROX_SIMPLE)
60  # canvas = np.zeros_like(image, np.uint8)
61  # for cnt in contour:
62  # cv2.drawContours(canvas, [cnt], 0, np.random.randint(1, 256), -1)
63  if len(contour) == 1:
64  mask = np.zeros_like(image, np.uint8)
65  cnt = contour[0]
66  cv2.drawContours(mask, [cnt], 0, 255, -1)
67 
68  x, y, w, h = cv2.boundingRect(contour[0])
69  box = np.array([x, y, x + w, y + h])
70  # self._logger.debug("Bounding box of segmentation is {}.".format(
71  # np.array_str(box, precision=2, suppress_small=True)))
72  elif len(contour) == 0:
73  self._logger.warning("No contour found!")
74  box = None
75  mask = None
76  else:
77  # self._logger.debug("Multiple contours found! Dilate image and repeat.")
78  img = np.zeros_like(image, np.uint8)
79  for cnt in contour:
80  cv2.drawContours(img, [cnt], 0, 255, -1)
81  kernel = cv2.getStructuringElement(shape=cv2.MORPH_ELLIPSE, ksize=(5, 5))
82  dilated = cv2.morphologyEx(img, cv2.MORPH_DILATE, kernel)
83  return self._morphology(image=dilated)
84  return contour, box, mask

86  def segment(self, image):
87  """Segment a single object in the given image.
88 
89  :param image: An image (numpy array) of shape (height, width, 3).
90  :return: A tuple containing
91  'box': The bounding box of the object; a (4,) numpy array.
92  'mask': The segmentation of the object; a (height, width) numpy
93  array.
94  """
95  gray = cv2.cvtColor(image, cv2.COLOR_RGB2GRAY)
96  equ = cv2.equalizeHist(gray)
97  _, thresh = cv2.threshold(equ, thresh=200, maxval=255, type=cv2.THRESH_BINARY)
98  contour, box, mask = self._morphology(image=thresh)
99  return box, mask
