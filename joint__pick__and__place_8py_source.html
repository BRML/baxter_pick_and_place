114  def _dither_pose(self, pose, fix_z=False):
115  """Modify the given pose slightly in a random fashion.
116 
117  :param pose: The pose to move to. A list of length 6
118  [x, y, z, roll, pitch, yaw].
119  :param fix_z: Whether to keep the z coordinate fixed.
120  :return: The slightly modified pose, a list of length 6
121  [x, y, z, roll, pitch, yaw].
122  """
123  borders = {
124  'x_min': pose[0] - 0.025,
125  'x_max': pose[0] + 0.025,
126  'y_min': pose[1] - 0.025,
127  'y_max': pose[1] + 0.025,
128  'z_min': pose[2],
129  'z_max': pose[2] + (0.025 if not fix_z else 0.0),
130  'roll_min': pose[3],
131  'roll_max': pose[3],
132  'pitch_min': pose[4],
133  'pitch_max': pose[4],
134  'yaw_min': pose[5] - np.deg2rad(5.0),
135  'yaw_max': pose[5] + np.deg2rad(5.0)
136  }
137  return self._robot.sample_pose(lim=borders)

139  def _move_to_pose_or_dither(self, arm, pose, fix_z=False):
140  """Shortcut to move the robots' specified limb to the given pose. If
141  the pose was not reached, modify the pose slightly and try again.
142 
143  :param arm: The arm <'left', 'right'> to control.
144  :param pose: The pose to move to. One of
145  - a ROS Pose,
146  - a list of length 6 [x, y, z, roll, pitch, yaw] or
147  - a list of length 7 [x, y, z, qx, qy, qz, qw].
148  :param fix_z: Whether to keep the z coordinate fixed.
149  :return: The achieved configuration, a dictionary of joint name keys
150  to joint angle values.
151  """
152  config = None
153  while config is None and not rospy.is_shutdown():
154  try:
155  config = self._robot.ik(arm=arm, pose=pose)
156  except ValueError:

163  def _calibrate_table_height(self):
164  """Calibrate the height of the table in the robot's task coordinates.
165  After ensuring that the table has been cleared of objects the robot
166  moves its limb to a number of randomly sampled poses in the task
167  space. At each pose it measures the distance to the table top using
168  an infrared sensor. The estimate of the table height is computed from
169  these distance measurements.
170 
171  :return: The estimated table height in meters.
172  """
173  setup_file = os.path.join(self._setup_dir, 'table_height.npz')
174  try:
175  with np.load(setup_file) as setup:
176  height = setup['height']
177  self._logger.info('Read table height %.3f m from calibration file.' % height)
178  except IOError:
179  self._logger.info('Calibrate table height.')
180  arm = 'left'
181  n_samples = 10
182  self._move_to_pose_or_raise(arm=arm, pose=settings.calibration_pose)
183  self._wait_for_clear_table(arm=arm)
184  heights = list()
185  for n in range(n_samples):
186  if rospy.is_shutdown():
187  break
188  config = None
189  while config is None:
190  if rospy.is_shutdown():
191  break
192  random_pose = self._robot.sample_task_space_pose(clip_z=True)
193  try:
194  config = self._robot.ik(arm=arm, pose=random_pose)
195  except ValueError:
196  pass
197  self._robot.move_to_config(config=config)
198  self.publish_vis(image=self._robot.cameras[arm].collect_image())
199  distances = list()
200  while len(distances) < 10:
201  d = self._robot.measure_distance(arm=arm)
202  if d is not None:
203  distances.append(d)
204  distance = np.mean(distances)
205  heights.append(self._robot.endpoint_pose(arm=arm)[2] -
206  (distance + self._robot.range_offset[2]))
207  heights = np.asarray(heights)
208  h_min = heights.min()
209  h_max = heights.max()
210  h_mean = heights.mean()
211  h_std = heights.std()
212  if h_std < 0.005:
213  height = h_mean
214  else:
215  height = h_max
216  msg = 'Computed table height to be %.3f m. ' % height
217  msg += '(min: {:.3f} m, max: {:.3f} m, mean: {:.3f} m, ' \
218  'std: {:.3f} m).'.format(h_min, h_max, h_mean, h_std)
219  self._logger.info(msg)
220  np.savez(setup_file, height=height)
221  self._robot.move_to_neutral(arm=arm)
222  self.publish_vis(image=255*np.ones((800, 1280, 3), dtype=np.uint8))
223  return float(height)

225  def _calibrate_table_view(self):
226  """Calibrate the left- and right hand camera view of the table.
227  After ensuring that the table has been cleared of objects the robot
228  records reference images of the empty table with the left and right
229  hand cameras and selects a number of adjacent patches on the table.
230 
231  :return: A dictionary containing
232  - the left hand camera reference image (key: 'image_left'),
233  - the right hand camera reference image (key: 'image_right'),
234  - the selected patches as a (n_patches, 2, 2) numpy array, where
235  the first dimension holds the patches and the second dimension
236  holds the top left and bottom right points defining the patch,
237  - the corresponding 2D positions on the table as a (n_patches, 3)
238  numpy array.
239  """
240  setup_file = os.path.join(self._setup_dir, 'table_view.npz')
241  try:
242  with np.load(setup_file) as setup:
243  data = {
244  'image_left': setup['image_left'],
245  'image_right': setup['image_right'],
246  'patches': setup['patches'],
247  'positions': setup['positions']
248  }
249  self._logger.info('Read table view from calibration file.')
250  except IOError:
251  self._logger.info('Calibrate table view.')
252  images = dict()
253  # define patches
254  (xl, yl), (xh, yh) = settings.table_limits
255  grid_x = range(xl, xh + 1, 95)
256  grid_y = range(yl, yh + 1, 95)
257  patches = list()
258  centers = list()
259  for i in range(len(grid_x) - 1):
260  for j in range(len(grid_y) - 1):
261  patches.append(((grid_x[i], grid_y[j]),
262  (grid_x[i + 1], grid_y[j + 1])))
263  centers.append(((grid_x[i + 1] - grid_x[i])//2 + grid_x[i],
264  (grid_y[j + 1] - grid_y[j])//2 + grid_y[j]))
265  positions = np.empty((len(patches), 3, 2), dtype=np.float32)
266  # record images and compute positions corresponding to patches
267  for i, arm in enumerate(['left', 'right']):
268  if rospy.is_shutdown():
269  break
270  self._move_to_pose_or_raise(arm=arm, pose=settings.calibration_pose)
271  self._wait_for_clear_table(arm=arm)
272  images[arm] = self._robot.cameras[arm].collect_image()
273  self.publish_vis(image=images[arm])
274  # illustrate patches on the table
275  canvas = images[arm].copy()
276  for patch, center in zip(patches, centers):
277  cv2.rectangle(canvas, pt1=patch[0], pt2=patch[1],
278  color=(0, 255, 0), thickness=1)
279  cv2.circle(canvas, center=center, radius=3,
280  color=(0, 255, 0), thickness=1)
281  self.publish_vis(image=canvas)
282 
283  for j, center in enumerate(centers):
284  if rospy.is_shutdown():
285  break
286  positions[j, :, i] = self._robot.estimate_object_position(arm=arm, center=center)
287 
288  self._robot.move_to_neutral(arm=arm)
289  data = {
290  'image_left': images['left'],
291  'image_right': images['right'],
292  'patches': np.array(patches),
293  'positions': positions.mean(axis=-1)
294  }
295  max_xyz = positions.std(axis=-1).max(axis=0)
296  if max_xyz.max() > 1e-3:
297  self._logger.warning("Estimated left and right 2D positions deviate "
298  "by up to {:.4f} m in x and {:.4f} m in y, "
299  "which is larger than 0.001 m!".format(
300  max_xyz[0], max_xyz[1]))
301  np.savez(setup_file, **data)
302  self.publish_vis(image=255*np.ones((800, 1280, 3), dtype=np.uint8))
303  return data

364  """Check whether a given pose lies within the robot's task space.
365 
366  :param pose: A position or pose [x, y, z, ...] (list of len >= 3).

373  zl = settings.task_space_limits_m['z_min']
374  zu = settings.task_space_limits_m['z_max']
375  if xl <= pose[0] <= xu and yl <= pose[1] <= yu and zl <= pose[2] <= zu:
376  return True
